int main()
{
    for(int i=1;i<=5;i++){  //row- outer loop
        for(int j=1;j<=i;j++){   //column- inne loop
            printf("Welcome\t");

}

#include <stdio.h>
int main()
{
    int n;
    scanf("%d",&n);
      for(int r=n;r>=1;r--){               //row- outer loop
      // printing space
        for(int c=1;c<=n-r;c++)        // r=beginning or current rows , n= no of rows  
            printf(" "); 
        for(int c=r;c>=1;c--)             // printing column
            printf("%d",c); 
         printf("\n");
      }
      
    return 0;
      }

//output
4
4321
 321
  21
   1
